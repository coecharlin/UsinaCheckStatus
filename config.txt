class GrowattApi:

    def getWeatherByPlantId(self, plantId):
        session.get(self.server_url)
        response = session.post(
            'http://server.growatt.com/index/getWeatherByPlantId',
            params={
                'plantId': plantId
            })
        data = json.loads(response.content.decode('utf-8'))
        return data
